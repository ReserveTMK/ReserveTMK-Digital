Build a complete Venue Hire Bookings & Automation system for Reserve Tāmaki with workflows, calendar booking, and flexible pricing.

CONTEXT:
We venue hire our space (rooms, hot desks) to individuals and organizations. Currently managing bookings manually with limited automation.

KEY REQUIREMENTS:
1. Automated workflow: Inquiry → Confirmed → Completed → Invoiced
2. Regular bookers can view calendar and select dates
3. Regular bookers can see remaining bookings (if prepaid packages)
4. Post-survey only for: one-off bookers OR first-time bookings
5. Manual survey trigger (send survey on demand)
6. Venue hire instructions + reminder sent together in confirmation
7. Regular booker pricing tiers: Full Paying, Community Discount, Koha/MOU

This is a 5-part implementation:
- Part 1: Database updates for workflows, pricing tiers, and booking packages
- Part 2: Accept/Decline workflow with automated emails (instructions + reminder combined)
- Part 3: Smart post-booking survey system (one-off or first-timers only)
- Part 4: Regular booker calendar + booking management
- Part 5: Xero integration with tiered pricing

-- Regular bookers with pricing tiers and booking packages
CREATE TABLE IF NOT EXISTS regular_bookers (
  id INT PRIMARY KEY AUTO_INCREMENT,
  contact_id INT NOT NULL,
  
  -- Organization details
  organization_name VARCHAR(200),
  billing_email VARCHAR(200) NOT NULL,
  billing_address TEXT,
  billing_phone VARCHAR(50),
  
  -- Pricing tier
  pricing_tier ENUM('full_paying', 'community_discount', 'koha_mou') NOT NULL DEFAULT 'full_paying',
  discount_percentage DECIMAL(5,2) DEFAULT 0.00, -- e.g., 20.00 for 20% off
  koha_mou_notes TEXT, -- Notes for koha/MOU arrangements
  
  -- Prepaid booking packages
  has_booking_package BOOLEAN DEFAULT FALSE,
  package_total_bookings INT DEFAULT 0, -- e.g., bought 10 bookings
  package_used_bookings INT DEFAULT 0, -- e.g., used 3 bookings
  package_remaining AS (package_total_bookings - package_used_bookings) STORED,
  package_expires_at DATE,
  
  -- Account access
  login_email VARCHAR(200),
  login_enabled BOOLEAN DEFAULT FALSE,
  can_view_calendar BOOLEAN DEFAULT TRUE,
  can_self_book BOOLEAN DEFAULT TRUE,
  
  -- Xero integration
  xero_contact_id VARCHAR(100),
  payment_terms ENUM('immediate', 'net_7', 'net_14', 'net_30') DEFAULT 'immediate',
  
  -- Status
  account_status ENUM('active', 'inactive', 'suspended') DEFAULT 'active',
  
  -- Notes
  notes TEXT,
  usual_booking_needs TEXT,
  
  -- Metadata
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  
  FOREIGN KEY (contact_id) REFERENCES contacts(id) ON DELETE CASCADE,
  INDEX idx_pricing_tier (pricing_tier),
  INDEX idx_account_status (account_status),
  INDEX idx_login_email (login_email)
);

-- Venue hire instructions (reusable templates)
CREATE TABLE IF NOT EXISTS venue_instructions (
  id INT PRIMARY KEY AUTO_INCREMENT,
  instruction_type ENUM('access', 'arrival', 'departure', 'emergency', 'general') NOT NULL,
  title VARCHAR(200),
  content TEXT,
  is_active BOOLEAN DEFAULT TRUE,
  display_order INT DEFAULT 0,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  INDEX idx_type (instruction_type),
  INDEX idx_active (is_active)
);

-- Surveys table (if not exists from Bundle 10)
CREATE TABLE IF NOT EXISTS surveys (
  id INT PRIMARY KEY AUTO_INCREMENT,
  survey_type ENUM('post_booking', 'mentoring_onboarding', '3_month', '6_month', 'exit') NOT NULL,
  related_id INT, -- booking_id for post_booking surveys
  contact_id INT, -- who we're surveying
  
  -- Survey data
  questions JSON,
  responses JSON,
  
  -- Status
  sent_at TIMESTAMP,
  completed_at TIMESTAMP,
  status ENUM('pending', 'sent', 'completed', 'expired') DEFAULT 'pending',
  
  -- Manual trigger tracking
  manually_triggered BOOLEAN DEFAULT FALSE,
  triggered_by INT, -- staff member who manually sent it
  
  -- Metadata
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  
  FOREIGN KEY (contact_id) REFERENCES contacts(id) ON DELETE CASCADE,
  FOREIGN KEY (triggered_by) REFERENCES contacts(id) ON DELETE SET NULL,
  INDEX idx_type (survey_type),
  INDEX idx_status (status),
  INDEX idx_related (related_id)
);

───────────────────────────────────────────────────────────────────────
SURVEY TRIGGER LOGIC (SMART)
───────────────────────────────────────────────────────────────────────

When booking marked as 'completed' OR manually triggered:

FUNCTION: should_send_automatic_survey(contact_id, booking_id)

-- Check 1: Is this a one-off booker?
SELECT COUNT(*) AS is_regular 
FROM regular_bookers 
WHERE contact_id = [contact_id] AND account_status = 'active';

-- Check 2: How many completed bookings does this contact have?
SELECT COUNT(*) AS previous_bookings
FROM bookings 
WHERE contact_id = [contact_id] 
AND status = 'completed'
AND id != [booking_id]; -- exclude current booking

-- Decision logic
IF is_regular == 0: 
  RETURN TRUE; -- One-off booker, always send survey

IF is_regular > 0 AND previous_bookings == 0:
  RETURN TRUE; -- Regular booker's first booking, send survey

IF is_regular > 0 AND previous_bookings > 0:
  RETURN FALSE; -- Regular booker with previous bookings, skip survey

-- Manual override
IF manually_triggered == TRUE:
  RETURN TRUE; -- Staff clicked "Send Survey Now", always send

END FUNCTION

  RETURN

STEP 2: Calculate line items based on tier
base_amount = hours * hourly_rate

IF pricing_tier == 'full_paying':
  line_items = [
    {
      "Description": "Space rental: [space] on [date] ([hours] hours at $[rate]/hr)",
      "Quantity": [hours],
      "UnitAmount": [hourly_rate],
      "AccountCode": "200",
      "TaxType": "OUTPUT2"
    }
  ]
  
IF pricing_tier == 'community_discount':
  discount_pct = get_discount_percentage(contact_id)
  discount_amt = base_amount * (discount_pct / 100)
  
  line_items = [
    {
      "Description": "Space rental: [space] on [date] ([hours] hours at $[rate]/hr)",
      "Quantity": [hours],
      "UnitAmount": [hourly_rate],
      "AccountCode": "200",
      "TaxType": "OUTPUT2"
    },
    {
      "Description": "Community discount ([discount_pct]%)",
      "Quantity": 1,
      "UnitAmount": -[discount_amt], // Negative amount
      "AccountCode": "200",
      "TaxType": "OUTPUT2"
    }
  ]
